class BasicStrategyAgent:
    """Model-based agent implementing basic Blackjack strategy."""

    def __init__(self, num_actions=2):
        self.use_raw = False

        # Create basic strategy table for hard hands
        # 0 = stand, 1 = hit
        self.hard_strategy = np.ones((22, 11), dtype=int)  # Default is hit

        # Stand on 17 and up regardless of dealer's card
        self.hard_strategy[17:, :] = 0

        # 12-16: stand against dealer 2-6, otherwise hit
        for i in range(12, 17):
            self.hard_strategy[i, 2:7] = 0

        # Create basic strategy table for soft hands (with usable ace)
        self.soft_strategy = np.ones((22, 11), dtype=int)  # Default is hit

        # Always stand on soft 19+
        self.soft_strategy[19:, :] = 0

        # Soft 18: stand against dealer 2-8, otherwise hit
        self.soft_strategy[18, 2:9] = 0

    def _extract_state_info(self, state):
        """Extract player score, dealer card, and usable ace info"""
        try:
            # Try to access as numpy array
            player_score = int(state[0]) if isinstance(state, np.ndarray) else None
            dealer_card = int(state[1]) if isinstance(state, np.ndarray) and len(state) > 1 else None
            has_usable_ace = bool(state[2]) if isinstance(state, np.ndarray) and len(state) > 2 else False

            # If extraction failed, try another method
            if player_score is None:
                # For simplicity, assume values 12-21 are common player scores
                player_score = 15  # Default middle value
                dealer_card = 10   # Default common dealer card
                has_usable_ace = False

            return player_score, dealer_card, has_usable_ace

        except Exception as e:
            # If all extraction methods fail, use default values
            print(f"State extraction error: {e}")
            return 15, 10, False  # Default values

    def step(self, state):
        """Return action based on basic strategy table."""
        player_score, dealer_card, has_usable_ace = self._extract_state_info(state)

        # Make sure dealer_card is in valid range (1-10)
        dealer_idx = min(dealer_card, 10)

        # Use appropriate strategy table
        if has_usable_ace and player_score <= 21:  # Soft hand
            action = self.soft_strategy[player_score][dealer_idx]
        else:  # Hard hand
            action = self.hard_strategy[player_score][dealer_idx]

        return int(action)

    def eval_step(self, state):
        """Same as step for evaluation but returns tuple for compatibility."""
        return self.step(state), None

  • Calculates true count (running count divided by remaining decks)
  • Adjusts basic strategy decisions based on count (more conservative with negative count, more aggressive with positive count)

class CardCountingAgent(BasicStrategyAgent):
    """Agent that extends basic strategy with card counting."""

    def __init__(self, num_actions=2):
        super().__init__(num_actions)

        # Card counting variables
        self.running_count = 0
        self.cards_seen = 0
        self.decks = 1  # Assuming 1 deck by default

    def step(self, state):
        """Take action using basic strategy but adjust based on count."""
        # Update card count based on visible cards
        self._update_card_count(state)

        # Get the base strategy action
        player_score, dealer_card, has_usable_ace = self._extract_state_info(state)

        # Get basic strategy action
        if has_usable_ace and player_score <= 21:  # Soft hand
            dealer_idx = min(dealer_card, 10)
            action = self.soft_strategy[player_score][dealer_idx]
        else:  # Hard hand
            dealer_idx = min(dealer_card, 10)
            action = self.hard_strategy[player_score][dealer_idx]

        # Calculate true count
        remaining_decks = max(self.decks - self.cards_seen / 52, 0.5)
        true_count = self.running_count / remaining_decks

        # Adjust basic strategy based on count
        # Higher count means more high cards (10s) are left - better for player
        if 12 <= player_score <= 16:
            # Borderline standing hands - stand more often with high count
            if true_count >= 2 and dealer_card in [2, 3, 4, 5, 6]:
                action = 0  # Stand
            elif true_count <= -2:
                action = 1  # Hit

        return int(action)

    def eval_step(self, state):
        """Same as step for evaluation but returns tuple for compatibility."""
        return self.step(state), None

    def _update_card_count(self, state):
        """Update card counting based on visible cards (Hi-Lo system)."""
        try:
            # Extract dealer's card
            _, dealer_card, _ = self._extract_state_info(state)

            # Hi-Lo counting: 2-6 (+1), 7-9 (0), 10-A (-1)
            if 2 <= dealer_card <= 6:
                self.running_count += 1
            elif dealer_card >= 10 or dealer_card == 1:  # 10-A
                self.running_count -= 1

            self.cards_seen += 1

            # Reset count if we've seen too many cards (deck reshuffled)
            if self.cards_seen > 52 * self.decks * 0.75:  # 75% of cards seen
                self.running_count = 0
                self.cards_seen = 0

        except Exception as e:
            # Silently continue if card counting fails
            pass

4. Model-Based Approach with MDP Processes

Purpose: Implements a mathematically rigorous approach to Blackjack using Markov Decision Process calculations.

  • Computes exact probabilities for all possible future states and outcomes
  • Calculates expected values for each action based on precise mathematical modeling
  • Makes decisions by selecting the action with the highest expected value
  • Tracks state transitions and outcome probabilities based on current deck composition
  • Recursively evaluates all possible dealer final hands and their probabilities
  • Incorporates card removal effects to adjust probability calculations as cards are seen
  • Uses dynamic programming techniques to efficiently solve the MDP
  • Provides mathematically optimal decisions rather than relying on heuristic rules

The MDP approach calculates the exact value of each state-action pair by considering all possible future scenarios weighted by their probabilities. This enables the agent to make theoretically optimal decisions even in complex or borderline situations where traditional basic strategy rules might be suboptimal.

class MDPBasicStrategyAgent:
    """Model-based agent implementing Blackjack basic strategy using MDP calculations."""

    def __init__(self, num_actions=2):
        self.use_raw = False
        self.num_decks = 1  # Default to 1 deck

        # For caching computed results
        self.expected_value_cache = {}

        # Initialize standard basic strategy for comparison
        self.hard_strategy = np.ones((22, 11), dtype=int)  # Default is hit
        self.hard_strategy[17:, :] = 0  # Stand on 17 and up regardless of dealer's card
        for i in range(12, 17):
            self.hard_strategy[i, 2:7] = 0  # 12-16: stand against dealer 2-6

        self.soft_strategy = np.ones((22, 11), dtype=int)  # Default is hit
        self.soft_strategy[19:, :] = 0  # Always stand on soft 19+
        self.soft_strategy[18, 2:9] = 0  # Soft 18: stand against dealer 2-8

    def _extract_state_info(self, state):
        """Extract player score, dealer card, and usable ace info"""
        try:
            # Try to access as numpy array
            player_score = int(state[0]) if isinstance(state, np.ndarray) else None
            dealer_card = int(state[1]) if isinstance(state, np.ndarray) and len(state) > 1 else None
            has_usable_ace = bool(state[2]) if isinstance(state, np.ndarray) and len(state) > 2 else False

            # If extraction failed, try another method
            if player_score is None:
                # Default values
                player_score = 15
                dealer_card = 10
                has_usable_ace = False

            return player_score, dealer_card, has_usable_ace

        except Exception as e:
            # Default values
            return 15, 10, False

    def _calc_card_probabilities(self, cards_seen=None):
        """Calculate probability of drawing each card value given cards seen so far.

        Args:
            cards_seen: Dictionary of {card_value: count} for cards already seen

        Returns:
            Dictionary of {card_value: probability} for drawing each card
        """
        if cards_seen is None:
            cards_seen = {}

        # Calculate remaining cards
        total_cards = 52 * self.num_decks
        remaining_cards = {
            # Card values (2-10, J, Q, K are worth face value, A is 1/11)
            # Count of each card in a deck (4 suits * number of decks)
            2: 4 * self.num_decks - cards_seen.get(2, 0),
            3: 4 * self.num_decks - cards_seen.get(3, 0),
            4: 4 * self.num_decks - cards_seen.get(4, 0),
            5: 4 * self.num_decks - cards_seen.get(5, 0),
            6: 4 * self.num_decks - cards_seen.get(6, 0),
            7: 4 * self.num_decks - cards_seen.get(7, 0),
            8: 4 * self.num_decks - cards_seen.get(8, 0),
            9: 4 * self.num_decks - cards_seen.get(9, 0),
            10: 16 * self.num_decks - cards_seen.get(10, 0),  # 10, J, Q, K are all worth 10
            1: 4 * self.num_decks - cards_seen.get(1, 0)  # Ace
        }

        # Calculate total remaining cards
        remaining_total = sum(remaining_cards.values())

        # Calculate probabilities
        probabilities = {card: count / remaining_total for card, count in remaining_cards.items()}

        return probabilities

    def _calc_dealer_final_probabilities(self, dealer_upcard):
        """Calculate probabilities of dealer's final hand totals given upcard.

        Args:
            dealer_upcard: Dealer's visible card (1-10)

        Returns:
            Dictionary of {final_total: probability}
        """
        # Cache key for memoization
        cache_key = f"dealer_{dealer_upcard}"
        if cache_key in self.expected_value_cache:
            return self.expected_value_cache[cache_key]

        # For simplicity, we'll use a recursive implementation
        def dealer_probs(current_total, has_usable_ace):
            # If dealer busts, return bust probability
            if current_total > 21:
                return {22: 1.0}  # Use 22 to represent bust

            # If dealer stands (17 or higher), return that total with 100% probability
            if current_total >= 17:
                return {current_total: 1.0}

            # If dealer must hit, calculate probabilities recursively
            card_probs = self._calc_card_probabilities()
            result = {}

            for card, prob in card_probs.items():
                # Calculate new total after drawing this card
                new_total = current_total + card
                new_has_ace = has_usable_ace

                # Handle ace logic
                if card == 1:  # Ace
                    if new_total + 10 <= 21:  # Can use ace as 11
                        new_total += 10
                        new_has_ace = True

                # Convert soft 21+ by making ace worth 1
                if new_has_ace and new_total > 21:
                    new_total -= 10
                    new_has_ace = False

                # Recursively get probabilities for this new state
                sub_probs = dealer_probs(new_total, new_has_ace)

                # Add these probabilities to result, weighted by this card's probability
                for total, subprob in sub_probs.items():
                    if total not in result:
                        result[total] = 0
                    result[total] += prob * subprob

            return result

        # Dealer's initial total is just the upcard
        initial_total = dealer_upcard
        has_ace = (dealer_upcard == 1)

        # If the upcard is an Ace, it counts as 11 initially
        if has_ace:
            initial_total = 11

        # Calculate and cache probabilities
        probs = dealer_probs(initial_total, has_ace)
        self.expected_value_cache[cache_key] = probs

        return probs

    def _calc_player_bust_probability(self, player_total, has_usable_ace):
        """Calculate probability of player busting if they hit.

        Args:
            player_total: Current player hand total
            has_usable_ace: Whether player has a usable ace

        Returns:
            Probability of busting
        """
        # If player has a soft hand, they can't bust with one card
        if has_usable_ace:
            return 0.0

        # Calculate which cards cause a bust
        bust_threshold = 21 - player_total

        # Get card probabilities
        card_probs = self._calc_card_probabilities()

        # Sum probabilities of cards that would cause a bust
        bust_prob = sum(prob for card, prob in card_probs.items() if card > bust_threshold)

        return bust_prob

    def _calc_expected_value(self, player_total, dealer_card, has_usable_ace, action):
        """Calculate expected value of taking an action in the current state.

        Args:
            player_total: Current player hand total
            dealer_card: Dealer's visible card
            has_usable_ace: Whether player has a usable ace
            action: 0 for stand, 1 for hit

        Returns:
            Expected value (average reward) for taking this action
        """
        # Cache key for memorization
        cache_key = f"ev_{player_total}_{dealer_card}_{has_usable_ace}_{action}"
        if cache_key in self.expected_value_cache:
            return self.expected_value_cache[cache_key]

        # Case 1: Player stands
        if action == 0:
            # Calculate dealer's final hand probabilities
            dealer_probs = self._calc_dealer_final_probabilities(dealer_card)

            # Calculate expected value
            ev = 0.0

            for dealer_total, prob in dealer_probs.items():
                # Dealer busts
                if dealer_total > 21:
                    ev += prob * 1.0  # Player wins
                # Player wins
                elif player_total > dealer_total:
                    ev += prob * 1.0
                # Push (tie)
                elif player_total == dealer_total:
                    ev += prob * 0.0
                # Dealer wins
                else:
                    ev += prob * (-1.0)

        # Case 2: Player hits
        else:
            # Get card probabilities
            card_probs = self._calc_card_probabilities()

            # Calculate expected value for each possible card drawn
            ev = 0.0

            for card, prob in card_probs.items():
                new_total = player_total + card
                new_has_ace = has_usable_ace

                # Handle ace logic
                if card == 1:  # Ace
                    if new_total + 10 <= 21:  # Can use ace as 11
                        new_total += 10
                        new_has_ace = True

                # Convert soft 21+ by making ace worth 1
                if new_has_ace and new_total > 21:
                    new_total -= 10
                    new_has_ace = False

                # If player busts, lose immediately
                if new_total > 21:
                    ev += prob * (-1.0)

                # If player gets 21, assume it's a win (simplified)
                elif new_total == 21:
                    ev += prob * 1.0

                # Otherwise, recursively calculate best action's expected value
                else:
                    # Calculate expected value of standing
                    stand_ev = self._calc_expected_value(new_total, dealer_card, new_has_ace, 0)

                    # Calculate expected value of hitting again
                    hit_ev = self._calc_expected_value(new_total, dealer_card, new_has_ace, 1)

                    # Take the better action
                    best_ev = max(stand_ev, hit_ev)
                    ev += prob * best_ev

        # Cache and return
        self.expected_value_cache[cache_key] = ev
        return ev

    def step(self, state):
        """Return action based on MDP calculations."""
        player_score, dealer_card, has_usable_ace = self._extract_state_info(state)

        # Calculate expected value of each action
        stand_ev = self._calc_expected_value(player_score, dealer_card, has_usable_ace, 0)
        hit_ev = self._calc_expected_value(player_score, dealer_card, has_usable_ace, 1)

        # Choose action with higher expected value
        if stand_ev >= hit_ev:
            return 0  # Stand
        else:
            return 1  # Hit

    def eval_step(self, state):
        """Same as step for evaluation but returns tuple for compatibility."""
        return self.step(state), None

class MDPCardCountingAgent(MDPBasicStrategyAgent):

        stand_ev = self._calc_expected_value(player_score, dealer_card, has_usable_ace, 0)
        hit_ev = self._calc_expected_value(player_score, dealer_card, has_usable_ace, 1)

        # Adjust expected values based on count (simplified adjustment)
        if true_count > 0:  # Positive count favors the player
            stand_ev += 0.01 * true_count  # Slight boost to standing EV
        elif true_count < 0:  # Negative count favors the dealer
            hit_ev += 0.01 * abs(true_count)  # Slight boost to hitting EV

        # Choose action with higher expected value
        if stand_ev >= hit_ev:

        else:

class BlackjackAgent:
    """Q-learning reinforcement learning agent using Q-learning."""

    def __init__(self, env_instance):
        self.env = env_instance
        self.q_table = {}

        # Learning parameters
        self.alpha = 0.1  # Learning rate
        self.discount_factor = 0.95
        self.epsilon = 1.0  # Exploration rate
        self.epsilon_decay = 0.995  # Epsilon decay rate for better exploration-exploitation balance
        self.min_epsilon = 0.01  # Minimum exploration rate

        # Statistics
        self.training_iterations = 0

    def get_state_key(self, state):
        """Convert state to a hashable key for the Q-table."""
        try:
            # Try to access elements (player_score, dealer_card) and (usable_ace ==> 1)
            player_score, dealer_card, usable_ace = state

            # Return a tuple that can be used as a dictionary key
            return (player_score, dealer_card, 1 if usable_ace else 0)
        except:
            # If the state structure is not what we expect, hash all of the state
            return hash(str(state))

    def choose(self, state):
        """Choose an action using epsilon-greedy strategy."""
        state_key = self.get_state_key(state)

        # Initialize Q-values for this state if not seen before
        if state_key not in self.q_table:
            self.q_table[state_key] = {a: 0.0 for a in range(self.env.action_space.n)}

        # Epsilon-greedy action selection
        if np.random.random() < self.epsilon:
            action = np.random.randint(0, self.env.action_space.n)
        else:
            action = max(self.q_table[state_key], key=self.q_table[state_key].get)

        return action

    def act(self, state):
        """Choose policy for evaluation."""
        state_key = self.get_state_key(state)

        # Initialize Q-values for this state if not seen before
        if state_key not in self.q_table:
            self.q_table[state_key] = {a: 0.0 for a in range(self.env.action_space.n)}

        # Greedy action with highest Q-value
        action = max(self.q_table[state_key], key=self.q_table[state_key].get)

        return action, None

    def feed(self, state, action, reward, next_state, done = False):
        """Learn based on experience."""
        try:
            state_key = self.get_state_key(state)

            # Initialize Q-values if needed
            if state_key not in self.q_table:
                self.q_table[state_key] = {a: 0.0 for a in range(self.env.action_space.n)}

            # Calculate target
            if done:
                target = reward
            else:
                next_state_key = self.get_state_key(next_state)

                # Initialize next state's Q-values if needed
                if next_state_key not in self.q_table:
                    self.q_table[next_state_key] = {a: 0.0 for a in range(self.env.action_space.n)}

                # Get the maximum Q-value for the next state
                target = reward + self.discount_factor * max(self.q_table[next_state_key].values())

            # Update Q-value
            self.q_table[state_key][action] = (1 - self.alpha) * self.q_table[state_key][action] + self.alpha * target

            # Decay epsilon for better exploration-exploitation balance
            if done:
                self.epsilon = max(self.min_epsilon, self.epsilon * self.epsilon_decay)

            self.training_iterations += 1

        except Exception as e:
            print("Error in feed function of update fails")
            print(e)
            pass

def train_agent(env, episodes=5000):
    q_agent = BlackjackAgent(env)

    rewards = []

    print(f"Training Q-learning agent for {episodes} episodes...")

    for episode in range(episodes):
        state, info = env.reset()
        total_reward = 0
        done = False

        # Quick evaluation every 1000 episodes
        if episode % 1000 == 0:
            rewards.append(evaluate_agent(env, q_agent))
            print(f"Episode {episode}/{episodes}")
            print(f"Current average reward: {rewards[-1]}")

        while not done:
            # Choose action using epsilon-greedy
            action = q_agent.choose(state)

            # Take action in environment
            next_state, reward, terminated, truncated, info = env.step(action)
            done = terminated or truncated

            # Feed experience to agent for learning
            q_agent.feed(state, action, reward, next_state, done)

            # Update state and reward
            state = next_state
            total_reward += reward

    # Final evaluation
    final_reward = evaluate_agent(env, q_agent)
    print(f"Q-Learning final average reward: {final_reward}")

    # Display some learned Q-values

# Function to evaluate the agent's performance
def evaluate_agent(env, agent, episodes=100):
    total_reward = 0

    for _ in range(episodes):
        state, info = env.reset()
        done = False
        episode_reward = 0

        while not done:
            action, _ = agent.act(state)
            next_state, reward, terminated, truncated, info = env.step(action)
            done = terminated or truncated
            episode_reward += reward
            state = next_state

        total_reward += episode_reward

    return total_reward / episodes

class HybridBlackjackAgent:
    """Combines model-based strategy with model-free learning and card counting."""

    def __init__(self):
        self.use_raw = False

        # Initialize component agents without arguments
        self.basic_strategy = BasicStrategyAgent()
        self.card_counting = CardCountingAgent()
        env = gym.make('Blackjack-v1')
        self.q_learning = BlackjackAgent(env)

        # For direct access to card counting state
        self.running_count = 0
        self.cards_seen = 0
        self.count_threshold = 2

    def _extract_state_info(self, state):
        """Extract player score, dealer card, and usable ace from state."""
        try:
            player_score, dealer_card, usable_ace = state
            return player_score, dealer_card, usable_ace
        except:
            # Use a default fallback if extraction fails
            return 15, 10, False

    def update_card_count(self, state):
        """Update internal card count based on dealer's visible card."""
        player_score, dealer_card, has_usable_ace = self._extract_state_info(state)

        # Hi-Lo counting: 2-6 (+1), 7-9 (0), 10-A (-1)
        if 2 <= dealer_card <= 6:
            self.running_count += 1
        elif dealer_card >= 10 or dealer_card == 1:  # Ace is often represented as 1
            self.running_count -= 1

        self.cards_seen += 1

    def step(self, state):
        """Take action using the appropriate strategy based on confidence."""
        # Update our internal card count
        self.update_card_count(state)

        # Calculate true count
        true_count = self.running_count / max(1, self.cards_seen / 52)

        # Decision logic for which strategy to use
        if self.q_learning.training_iterations < 5000:
            # Early in training, rely on model-based approach
            action, _ = self.card_counting.act(state)
            return action
        elif abs(true_count) >= self.count_threshold:
            # When count is significant, use model-free approach
            return self.q_learning.act(state)[0]
        else:
            # Default to card counting enhanced basic strategy
            action, _ = self.card_counting.act(state)
            return action

    def act(self, state):
        """Same as step but returns tuple for compatibility."""
        return self.step(state), None

    def eval_step(self, state):
        """Same as step for evaluation but returns tuple for compatibility."""
        return self.step(state), None

    def feed(self, state, action, reward, next_state, done=False):
        """Feed transition to the Q-learning component."""
        # Only the Q-learning component learns from experience
        self.q_learning.feed(state, action, reward, next_state, done)
